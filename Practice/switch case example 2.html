Switch Case using prompt tag

        let char = prompt("Enter a character: ")
        switch(char){
            case "a": {
                console.log("a")
                break
            }
            case "e": {
                console.log("e")
                break
            }
            case "i": {
                console.log("o")
                break
            }
            case "o": {
                console.log("o")
                break
            }
            case "u": {
                console.log("u")
                break
            }
            case "A": {
                console.log("A")
                break
            }
            case "E": {
                console.log("E")
                break
            }
            case "I": {
                console.log("I")
                break
            }
            case "O": {
                console.log("O")
                break
            }
            case "U": {
                console.log("U")
                break
            }
        }
