// 1. for Loop
for (let i = 1; i <= 5; i++) {
  console.log("Iteration " + i);
}

// 2. while Loop
let j = 1;
while (j <= 5) {
  console.log("Iteration " + j);
  j++;
}

// 3. do...while Loop
let k = 1;
do {
  console.log("Iteration " + k);
  k++;
} while (k <= 5);

// 4. for...of Loop (for Arrays)
const numbers = [10, 20, 30, 40, 50];
for (let num of numbers) {
  console.log(num);
}
